def initialize(resp)
  @limit = resp["X-Rate-Limit-Limit"].to_i
  @remaining = resp["X-Rate-Limit-Remaining"].to_i
  @reset = Time.at(resp["X-Rate-Limit-Reset"].to_i)
end
